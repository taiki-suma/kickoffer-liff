<script>

    // --- Gemini API Integration ---

    const askAiButton = document.getElementById('ask-ai-button');

    const questionInput = document.getElementById('ai-question');

    const answerContainer = document.getElementById('ai-answer-container');


    askAiButton.addEventListener('click', async () => {

        const userQuestion = questionInput.value;

        if (!userQuestion) {

            answerContainer.textContent = '質問内容を入力してください。';

            return;

        }


        // Show loading indicator

        answerContainer.innerHTML = '<div class="loader"></div>';


        // --- Knowledge Base for the AI ---

        // This is a simple knowledge base. In a real application, this might come from a database or a CMS.

        const knowledgeBase = `

        - Kick Offerは、ジュニアユース年代のサッカー選手とチームを繋ぐマッチング支援サービスです。

        - 選手の利用は完全に無料です。

        - チームを探すには、公式アプリの利用を推奨しています。アプリはApp StoreとGoogle Playからダウンロードできます。

        - 気になるチームが見つかったら、LINEを通じて直接チームの担当者と連絡を取ることができます。運営が間に入って安全に繋ぎます。

        - 練習会への参加費用は、各チームの規定によります。詳細はチームに直接お問い合わせください。

        - 運営への直接の問い合わせも可能です。その場合は「運営へ質問」とメッセージを送ってください。

        - 対象地域は現在、一都三県（東京都、神奈川県、埼玉県、千葉県）を中心に拡大中です。

        `;


        // --- Prompt Engineering ---

        // Create a clear prompt for the Gemini model.

        const prompt = `あなたはジュニアユースのチーム探しを支援する「Kick Offer」の優秀なアシスタントです。以下の知識ベースの情報だけを参考にして、ユーザーの質問に親切かつ簡潔に日本語で答えてください。知識ベースにない質問の場合は、「そのご質問については分かりかねます。運営にお繋ぎしますので『運営へ質問』と送信してください。」と回答してください。\n\n[知識ベース]\n${knowledgeBase}\n\n[ユーザーの質問]\n${userQuestion}`;


        try {

            // Call the Gemini API

            let chatHistory = [];

            chatHistory.push({ role: "user", parts: [{ text: prompt }] });

            const payload = { contents: chatHistory };

            const apiKey = ""; // APIキーはCanvas環境で自動的に設定されます

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

            

            const response = await fetch(apiUrl, {

                method: 'POST',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify(payload)

            });


            if (!response.ok) {

                throw new Error('APIからの応答がありません。');

            }

            

            const result = await response.json();


            // Display the result

            if (result.candidates && result.candidates.length > 0 &&

                result.candidates[0].content && result.candidates[0].content.parts &&

                result.candidates[0].content.parts.length > 0) {

                const text = result.candidates[0].content.parts[0].text;

                answerContainer.textContent = text;

            } else {

                 answerContainer.textContent = '回答を生成できませんでした。もう一度お試しください。';

            }


        } catch (error) {

            console.error('Error calling Gemini API:', error);

            answerContainer.textContent = 'エラーが発生しました。時間をおいて再度お試しください。';

        }

    });

</script>
